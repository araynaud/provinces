/** @module common/components/sldsCheckbox
@description A checkable input that communicates if an option is true, false or indeterminate. This component displays a checkbox with SLDS style.
<center>
	<img style="height: 150px" src="https://user-images.githubusercontent.com/21251436/37166265-75ac1c52-22b3-11e8-8bda-6b1079cefd4a.png"/>
</center>

@see {@link https://www.lightningdesignsystem.com/components/checkbox/#content|Checkbox - Lightning Design System}

@param model           {Boolean}                 =  model value
@param readOnly        {Boolean}                 =  read-only /disabled flag
@param required        {Boolean}                 =  required flag
@param variable        {optional|String}         @  global variable name
@param checkboxId      {optional|String}         @  an id to set on checkbox input
@param checkboxName    {optional|String}         @  a name to set on checkbox input
@param label           {optional|String}         @  a label to display right after the checkbox
@param hideLabel       {optional|String}         <  hide the checkbox label
@param questionFormat  {optional|String}         @  classes for label format: "text-format-bold | text-format-underlined | text-format-italic"
@param tooltip         {optional|String}         @  tooltip / popover text
@param onChange        {optional|function}       &  a callback function called when model changes

@example
<slds-checkbox checkbox-name="{{inputName}}" variable="sldsCheckbox.{{field}}_{{row.$rowindex}}" model="row[field]" required="vm.isRequired(col)" hide-label="true" read-only="vm.isReadOnly(col,row)">
</slds-checkbox>

*/

module.exports = function (ngModule)
{
	ngModule.component('sldsCheckbox',{
		bindings: {
			model: '=',
			readOnly: '=',
			required: '=',
			variable: '@',
			label: '@',
			checkboxId: '@',
			checkboxName: '@',
			tooltip: '@',
			cssStyle: '=',
			onChange: '&',
			hideLabel: '<',
			questionFormat:'@'
		},
		template: require('./sldsCheckbox.html'),
		controllerAs: 'vm',
		controller: SldsCheckboxController
	});

	SldsCheckboxController.$inject = ['$timeout', 'CommonUtils'];
	function SldsCheckboxController  ( $timeout,   CommonUtils )
	{
		var vm = this;
		CommonUtils.setVar(vm.variable, vm);

		vm.model = Boolean(vm.model);

		if(angular.isFunction(vm.onChange))
			vm.changeEvent = () => $timeout(vm.onChange, 0);
	}
};
