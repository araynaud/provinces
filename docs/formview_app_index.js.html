/** @module formview

@description
A form generator based on custom version of angular-formly using the Salesforce Lightning Design System.
The index.js file declares the core angular app and configures the custom field types for angular-formly
*/

// pass the generic module reference to all required components and controllers.
var ngModule = angular.module('app', ['formly', 'ngSanitize', 'common'], ["formlyConfigProvider", function(formlyConfigProvider)
{
	formlyConfigProvider.setType({ name: 'text',            template: require('./components/qTemplates/text.html') });
	formlyConfigProvider.setType({ name: 'long text',       template: require('./components/qTemplates/long-text.html') });
	formlyConfigProvider.setType({ name: 'email address',   template: require('./components/qTemplates/email-address.html') });
	formlyConfigProvider.setType({ name: 'number',          template: require('./components/qTemplates/number.html') });
	formlyConfigProvider.setType({ name: 'min/max scale',   template: require('./components/qTemplates/range.html') });
	formlyConfigProvider.setType({ name: 'date',            template: require('./components/qTemplates/date.html') });
	formlyConfigProvider.setType({ name: 'datetime',        template: require('./components/qTemplates/datetime.html') });
	formlyConfigProvider.setType({ name: 'checkbox',        template: require('./components/qTemplates/checkbox.html') });
	formlyConfigProvider.setType({ name: 'radio',           template: require('./components/qTemplates/radio.html') });
	formlyConfigProvider.setType({ name: 'picklist',        template: require('./components/qTemplates/picklist.html') });
	formlyConfigProvider.setType({ name: 'multi-picklist',  template: require('./components/qTemplates/multi-picklist.html') });
	formlyConfigProvider.setType({ name: 'search',          template: require('./components/qTemplates/search.html') });
	formlyConfigProvider.setType({ name: 'multi-search',    template: require('./components/qTemplates/search.html') });
	formlyConfigProvider.setType({ name: 'upload',          template: require('./components/qTemplates/upload.html') });

	//label: no field for answer
	formlyConfigProvider.setType({ name: 'label',           template: require('./components/qTemplates/label.html') });
	formlyConfigProvider.setType({ name: 'spacer',          template: require('./components/qTemplates/spacer.html') });

	//query questions
	formlyConfigProvider.setType({ name: 'query requestor', template: require('./components/qTemplates/query.html') });
	formlyConfigProvider.setType({ name: 'query responder', template: require('./components/qTemplates/query.html') });

	// set template for Nutritional info
	formlyConfigProvider.setType({ name: 'nutritional',                  template: require('./components/qTemplates/nutritional.html') });
	formlyConfigProvider.setType({ name: 'nutritional-2017',             template: require('./components/qTemplates/nutritional.html') });
	formlyConfigProvider.setType({ name: 'nutritional-100g',             template: require('./components/qTemplates/nutritional.html') });
	formlyConfigProvider.setType({ name: 'nutritional-100g-2017',        template: require('./components/qTemplates/nutritional.html') });

	//product test questions including results, executive summary, failures, selection
	formlyConfigProvider.setType({ name: 'product test',                 template: require('./components/qTemplates/productTest.html') });
	formlyConfigProvider.setType({ name: 'product test results',         template: require('./components/qTemplates/productTest.html') });
	formlyConfigProvider.setType({ name: 'product test selection',       template: require('./components/qTemplates/productTestSelection.html') });
	formlyConfigProvider.setType({ name: 'summary of failures',          template: require('./components/qTemplates/summaryOfFailures.html') });
	formlyConfigProvider.setType({ name: 'executive summary',            template: require('./components/qTemplates/executiveSummary.html') });
	formlyConfigProvider.setType({ name: 'lab selection',                template: require('./components/qTemplates/lab-selection.html') });
	formlyConfigProvider.setType({ name: 'product test - test deletion', template: require('./components/qTemplates/multi-picklist.html') });
}]);
