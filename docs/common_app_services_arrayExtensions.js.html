//global variable test functions
	function isMissing(variable)
	{
		return typeof variable === "undefined" || variable === null;
	}
	String.isMissing = Array.isMissing = Object.isMissing = isMissing;

	function isDefined(variable, ctx)
	{
		if(!ctx) ctx = window;
		if(!variable) return ctx;

		var arr = variable.split(".");
		for(var i = 0; i < arr.length; i++)	{
			if(isMissing(ctx[arr[i]])) return false;
			ctx = ctx[arr[i]];
		}
		return true;
	}

	function valueIfDefined(variable, ctx)
	{
		if(!ctx) ctx = window;
		if(!variable) return ctx;

		var arr = Array.isArray(variable) ? variable : variable.split(".");
		if(arr.length == 1) //if no . return directly
			return ctx[variable];

		for(var i = 0; i < arr.length; i++) {
			if(isMissing(ctx[arr[i]])) return ctx[arr[i]];
			ctx = ctx[arr[i]];
		}
		return ctx;
	}

	function valueOrDefault(variable, defaultValue)
	{
		return isMissing(variable) ? defaultValue : variable;
	}

	function isObjectNotArray(input)
	{
		return angular.isObject(input) && !angular.isArray(input);
	}

	/** @method isEmpty
@description tests if a variable is undefined or null, or Array has no elements, or string is Empty, or Object has no fields.
@param variable {any} variable to test
@return {boolean} true if variable is empty, false otherwise
*/

	function isEmpty(variable)
	{
		if(isMissing(variable)) return true;
		if(!isMissing(variable.length)) return variable.length === 0;
		if(angular.isObject(variable))	return !Object.hasKeys(variable);
		return false;
	}
	String.isEmpty = Array.isEmpty = Object.isEmpty = isEmpty;

@description tests if a variable is defined, and Array has elements, or string is not Empty, or Object has fields.
@param variable {any} variable to test

@see module:String~isEmpty

	function isNotEmpty(variable)

	/** @method plural
@description Add plural to string depending on number
@param nb {Number} number
@param str {String} singular word
@param str {optional|String} plural word if different (ex: 'child', 'children') (default: word + 's')

@return {String} pluralized string "No results", "1 result", "3 results" ...
*/

	String.plural = function(nb, word, pluralWord)
	{
		if(!word) return nb;

		if(nb == 1 || nb == -1)
			return nb + " " + String.toSingular(word);

		if(!nb)
			nb = valueIfDefined("labels.common.zero") || "No";

		return nb + " " + (pluralWord || String.toPlural(word));
	};

	String.toPlural   = word => String.pluralize(word, true);
	String.toSingular = word => String.pluralize(word, false);

	String.pluralize = function(word, isPlural)
	{
		if(String.isEmpty(word)) return word;

		word = word.trim();
		//if multiple words: process each word recursively
		if(word.includes(" ")) {
			let words = String.split(word, " ", true, true);
			words = words.map(w => String.pluralize(w, isPlural));
			return words.join(" ");
		}

		//simple word plural: add a s if not already ending with s
		if(!word.includes("*"))
			return isPlural && word[word.length - 1] != "s" ? word + "s" : word;

		//base*plural or base*singular*plural
		let fragments = String.split(word, "*", false, true);
		if(fragments.length <= 1) return fragments[0];

		let pluralEnd = fragments.last();
		let singularEnd = fragments.length >= 3 ? fragments[1] : "";
		return fragments[0] + (isPlural ? pluralEnd : singularEnd);
	};

	//if empty return empty array remove empty strings, trim other ones
	String.split = function(str, sep, filter, trim)
	{
		if(String.isEmpty(str)) return [];
		var fragments = str.split(sep);
		if(trim)    fragments = fragments.map(s => s.trim());
		if(filter)  fragments = fragments.filter(String.isNotEmpty);
		return fragments;
	};

	/** @method includes / includesIgnoreCase
@description tests if a String contains with another

@param caseSensitive {optional|boolean} if true: case sensitive, othwerwise: case insensitive

	String.includes = function(str, value, caseSensitive)
	{
		if(String.isEmpty(str) && String.isEmpty(value))
			return true;
		if(String.isEmpty(str))
			return false;
		if(caseSensitive)

		return str.toString().includesIgnoreCase(value);
	};

		return this.toLowerCase().includes(value.toString().toLowerCase());

		return (indexOf >= 0) ? (indexOf + startpos) : indexOf;

	/** @method regexLastIndexOf
@description Get position of the last occuence of a regular expession inside a String.
@param regex         {String}  Regex to find in str

@return {String} if substring found: 0-based index
<br/>if substring not found: -1
<br/>
*/

	String.prototype.regexLastIndexOf = function(regex, startpos)
	{
		regex = (regex.global) ? regex : new RegExp(regex.source, "g" + (regex.ignoreCase ? "i" : "") + (regex.multiLine ? "m" : ""));

		if(typeof (startpos) == "undefined")
			startpos = this.length;
		else if(startpos < 0)
			startpos = 0;

		var stringToWorkWith = this.substring(0, startpos + 1);
		var lastIndexOf = -1;
		var nextStop = 0;
		var result;
		while((result = regex.exec(stringToWorkWith)) != null) {
			lastIndexOf = result.index;
			regex.lastIndex = ++nextStop;
		}
		return lastIndexOf;
	};

	/** @method substringBefore
@description Get portion of a string before specified substring, if found.
@param s             {String}  input string to test
@param sub           {String}  Substring to find in str
@param last          {Boolean} true: last occurence of sub / default: first occurence of sub
@param stringOrEmpty {Boolean} result if substring not found. true: empty string / default: whole input string
@param include       {Boolean} true: include sub in result / default: do not include sub in result
@param ignoreCase    {Boolean} true: case-insentitive search for sub / default: case-sensitive

@return {String} if substring found: portion of input string before specified substring
<br/>if substring not found: stringOrEmpty == true: empty string  / default: whole input string
<br/>
*/

	String.substringBefore = function(s, sub, last, stringOrEmpty, include, ignoreCase)
	{
		if(String.isEmpty(s)) return String.convert(s);

		var lstr = s   = String.convert(s);
		var lsub = sub = String.convert(sub);
		if(ignoreCase) {
			lstr = lstr.toLowerCase();
			lsub = lsub.toLowerCase();
		}

		var pos = last ? lstr.lastIndexOf(lsub) : lstr.indexOf(lsub);
		if(pos < 0)
			return stringOrEmpty ? "" : s;

		if(include)
			pos += sub.length;
		return s.substring(0, pos);
	};

	String.prototype.substringBefore = function(sub, last, stringOrEmpty, include, ignoreCase) {
		return String.substringBefore(this, sub, last, stringOrEmpty, include, ignoreCase);
	};

	/** @method substringAfter
@description Get portion of a string after specified substring, if found.
@param s             {String}  input string to test
@param sub           {String}  Substring to find in str
@param last          {Boolean} true: last occurence of sub / default: first occurence of sub
@param stringOrEmpty {Boolean} result if substring not found. true: whole input string / default: empty string
@param include       {Boolean} true: include sub in result / default: do not include sub in result

@return {String} if substring found: portion of input string after specified substring.
<br/>if substring not found: stringOrEmpty == true: whole input string / default: empty string
<br/>
*/

	String.substringAfter = function(s, sub, last, stringOrEmpty, include, ignoreCase)
	{
		if(String.isEmpty(s)) return String.convert(s);

		var lstr = s   = String.convert(s);
		var lsub = sub = String.convert(sub);
		if(ignoreCase) {
			lstr = lstr.toLowerCase();
			lsub = lsub.toLowerCase();

		var pos = last ? lstr.lastIndexOf(lsub) : lstr.indexOf(lsub);
		if(pos < 0)
			return stringOrEmpty ? s : "";

		if(!include)
			pos += sub.length;

		return s.substring(pos);
	};

	String.prototype.substringAfter = function(sub, last, stringOrEmpty, include, ignoreCase) {
		return String.substringAfter(this, sub, last, stringOrEmpty, include, ignoreCase);

	/** @method compare
@description Comparison function used in Array sorting functions
@param a {String|any} input variable
@param b {String|any} input variable
@param caseSensitive {optional|boolean} if true: case sensitive, othwerwise: case insensitive
@param useLocale {optional|boolean} if true: use browser locale
@return {Number} 0 if a == b
<br/>1 if a > b
<br/>-1 if a < b
*/
	String.compare = function(a, b, caseSensitive, useLocale)
	{
		a = String.convert(a);
		b = String.convert(b);

		if(!caseSensitive && useLocale)
		{
			a = a.toLocaleLowerCase();
			b = b.toLocaleLowerCase();
		}
		else if(!caseSensitive)
		{
			a = a.toLowerCase();
			b = b.toLowerCase();
		}

		if(a > b) return 1;
		if(a < b) return -1;
		return 0;
	};

	//append 2 string with separator in between
	String.appendString = function(input, separator, str)
	{
		if(arguments.length == 2) {
			str = separator;
			separator = '';
		}

		if(isMissing(input)) input = '';
		if(isMissing(str))   str = '';
		if(isEmpty(str))   return input;
		if(isEmpty(input)) return str;

		if(isMissing(separator)) separator = '';
		return input + separator + str;
	};

	String.prototype.appendString = function(separator, str)
	{
		if(arguments.length == 1)
			return String.appendString(this.toString(), separator);

		return String.appendString(this.toString(), separator, str);
	};

	function getFilterFunction(funct)
	{
		if(!funct) {
			funct = e => !!e;
		}
		else if(angular.isString(funct) || angular.isArray(funct)) {
			var field = funct;
			funct = el => valueIfDefined(field, el);
		}
		return funct;
	}

	Array.some = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return false;
		funct = getFilterFunction(funct);
		return arr.some(funct);
	};

	Array.every = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return false;
		funct = getFilterFunction(funct);
		return arr.every(funct);
	};

	//check if all elements return function = false
	Array.none = function(arr, funct) {
		return !Array.some(arr, funct);
	};

	Array.find = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return;

		if(!Array.isArray(arr))
			arr = Object.values(arr);

		funct = getFilterFunction(funct);
		return arr.find(funct);
	};

	Array.findIndex = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return -1;

		funct = getFilterFunction(funct);
		return arr.findIndex(funct);
	};

	Array.filter = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return [];

		if(!Array.isArray(arr))
			arr = Object.values(arr);

		funct = getFilterFunction(funct);
		return arr.filter(funct);
	};

	Array.count = function(arr, funct) {
		return Array.isEmpty(arr) ? 0 : Array.filter(arr, funct).length;
	};

	//map array by function or field name
	Array.map = function(arr, funct)
	{
		if(Array.isEmpty(arr)) return [];
		if(!funct) return arr;

		if(!Array.isArray(arr))
			arr = Object.values(arr);

		funct = getFilterFunction(funct);
		return arr.map(funct);
	};

	Array.includes = function(arr, element)

	{
		if(!angular.isArray(arr)) return false;

		if(ignoreCase && angular.isString(element))
			element = element.toLowerCase();
		if(ignoreCase)
			arr = arr.map(str => angular.isString(str) ? str.toLowerCase() : str);

		return arr.includes(element);
	};

	//remove multiple elements from array

	Array.removeMultiple = function(arr, elements, field)
	{
		if(Array.isEmpty(elements) || Array.isEmpty(arr))
			return arr;

		for(var i = 0; i < elements.length; i++)
		{
			if(Array.isEmpty(arr))
				return arr;
			Array.remove(arr, elements[i], field);
		}
		return arr;
	};

	//diff function that does not modify input array
	Array.diff = function(arr, elements, field)
	{
		var result = [];
		for(var i = 0; i < Array.getCount(arr); i++)
		{
			var index = Array.findBy(elements, arr[i], field);
			if(index == -1)
				result.push(arr[i]);
		}
		return result;
	};

	Array.prototype.diff = function(elements, field) {
		return Array.diff(this, elements, field);
	};

	//return index of element where field = value
	Array.findBy = function(arr, value, field)
	{
		if(Array.isEmpty(arr))
			return -1;

		var searchValue = value;
		if(value && value.hasOwnProperty(field))
			value = value[field];
		else if(value && field)
		{
			var searchFieldValue = valueIfDefined(field, value);
			if(!isMissing(searchFieldValue))
				searchValue = searchFieldValue;
		}

		for(var i = 0; i < arr.length; i++)
		{
			if(arr[i] === value)
				return i;

			var elementValue = valueIfDefined(field, arr[i]);
			if(elementValue === value || elementValue === searchValue)
				return i;
		}

		return -1;
	};

	Array.prototype.shuffle = function()

			var randomElement = this.splice(Math.random() * this.length, 1)[0];

	//divide into pages with fixed number. nb = max number per page
	Array.prototype.paginate = function(nb, equalize, minPages)
	{
		if(isEmpty(this))
			return [];
		minPages = valueOrDefault(minPages, 1);
		var nbPages = Math.ceil(this.length / nb);
		if(isMissing(nb) || nb <= 1 || nb > this.length || nbPages < minPages)
			return [this];

		var result = [];
		var remaining = this.length;
		var start = 0;

		for(var i = 0; i < nbPages; i++)
		{
			if(equalize)
				nb = Math.ceil(remaining / (nbPages - i));
			result.push(this.slice(start, start + nb));
			start += nb;
			remaining -= nb;
		}
		return result;
	};

	//divide array into N slices

	Array.prototype.divideInto = function(nb, transpose)
	{
	//divide in 1 or more than length = same array
		if(!this.length)
			return [];
		if(nb <= 1)
			return [this];
		nb = Math.floor(nb);
		if(nb > this.length)
			nb = this.length;
		var result = [];
		var i = 0;
		if(transpose)
		{
			var remainingElements = this;
			for(i = 0; i < nb; i++)
			{
				var perCol = remainingElements.length / (nb - i);
				perCol = (i + 1 <= nb / 2) ? Math.ceil(perCol) : Math.floor(perCol);
				result[i] = remainingElements.slice(0, perCol);
				remainingElements = remainingElements.slice(perCol);
			}
		}
		else
		{
			for(i = 0; i < nb; i++)
				result.push([]);
			for(i = 0; i < this.length ; i++)
				result[i % nb].push(this[i]);
		//result[nb - 1].push(this[i]);
		}
		return result;
	};

	function compareElements(a, b, reverse, caseSensitive)
	{
		if(angular.isObject(a)) a = Object.getFirstValue(a);
		if(angular.isObject(b)) b = Object.getFirstValue(b);

		if(String.isEmpty(a)) a = "";
		else if(!caseSensitive && angular.isString(a)) a = a.toUpperCase();

		if(String.isEmpty(b)) b = "";
		else if(!caseSensitive && angular.isString(b)) b = b.toUpperCase();

		var rev = reverse ? -1 : 1;
		if(a > b) return rev;
		if(a < b) return -rev;
		return 0;
	}
	Array.compareElements = compareElements;

	//convert Array of Objects to CSV format string
	Array.toCsv = function(data, columns, separator)
	{
		if(Array.isEmpty(data))
			return "";

		if(!separator)
			separator = "\t";

		if(angular.isString(columns))
			columns = String.split(columns, /[,;]/, true, true);

		var headers = Array.isNotEmpty(columns);
		//get all possible columns from every row in order they appear
		if(Array.isEmpty(columns) || !Array.isArray(columns))
			columns = data.distinct(Object.keys).join(",").split(",").distinct();

		var rows = data.map(row => columns.map(col => {
			var value = valueOrDefault(valueIfDefined(col, row), "");
			return Array.isArray(value) ? value.length : value;
		}).join(separator) );

		if(headers)
			rows.unshift(columns.join(separator));
		rows.push("");
		return rows.join("\n");
	};

	//convert CSV format string to Array of Arrays / Array of Objects

	// String[] : use field names for columns.  return Array of Objects
	// if true: take 1st row of csvData as header field list
	// if false/undefined : do not use headers. return Array of Arrays
	Array.fromCsv = function(csvData, columns, separator)
	{
		if(Array.isEmpty(csvData))
			return [];

		if(!separator)
			separator = "\t";

		var rows = String.split(csvData, /\n|\r/, true, true);
		rows = rows.map(row => row.trimEnd().split(separator));

		if(columns === true)
			columns = rows.shift(); //use first row as column headers
		else if(angular.isString(columns))
			columns = String.split(columns, /[,;]/, false, true);
		if(Array.isNotEmpty(columns))
			rows = rows.map(row => Array.combine(row, columns, true) );

		return rows;
	};
