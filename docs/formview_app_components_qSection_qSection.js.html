/** @module formview/components/qSection
@description This component defines the section containers
@param {Array} sections - An array of objects that represents all section in the current tab.
@param {Object} section - The object of the current section.
@param {Object} model - The model of form values
@example Usage:
<q-section ng-repeat="section in vm.tab.sections track by section.id" section="section" model="vm.model" ><q-section>
*/
module.exports = function (ngModule)
{
	'use strict';
	ngModule.component('qSection', {
		bindings: {
			section: '=',
			model: '='
		},
		template: require('./qSection.html'),
		controllerAs: 'vm',
		controller: qSection
	});

	qSection.$inject = ['SldsTableService', 'TabService'];
	function qSection  ( SldsTableService,   TabService )
	{
		var vm = this;

		vm.$onInit = function()
		{
			vm.isInfo = vm.isInfoSection(vm.section);
			vm.settings = window.formViewSettings;
			vm.replaceLinks = TabService.replaceLinks;

			//set remote action methods for search fields in tables
			if(!vm.section || !vm.section.type != "Table" || Array.isEmpty(vm.section.elements)) return;

			angular.forEach(vm.section.elements, function(element)
			{
				if(!vm.isLookup(element)) return;

				element.modelField = "value";
				element.modelJoin = ";";
				element.minLength = 2;

				//multi-picklist: list of options pre-loaded, no remote action
				if(element.type == 'multi-picklist')
					element.minLength = 0;

				//lookups: list of options loaded via remote actions
				//only if optionListName specified
				else if(element.templateOptions && element.templateOptions.optionListName)
				{
					element.remoteActionArgs = element.templateOptions.optionListName;
					element.initAction = "getItemsByValue";
					element.searchAction = "getItemList";
				}
			});
		};

		vm.deleteTableRowAnswers = function(row, index)
		{
			if(!angular.isDefined(index))
				index = row.$rowindex;

			SldsTableService.deleteTableRowAnswers(row, index).then(function(result) {
				console.log('Table row deleted, index - ', index);
			});
		};

		//add / delete group section row
		vm.deleteSectionRow = index => TabService.deleteSectionRow(vm.section, index);
		vm.addSectionRow    = index => TabService.addSectionRow(vm.section, index);

		vm.logFormSection = function()
		{
			console.log("formSection =", window.formSection = vm.section);
			if(vm.model.tables && vm.model.tables[vm.section.id])
				console.log("sectionModel =", window.sectionModel = vm.model.tables[vm.section.id]);
		};

		/** @method isLookup
		@description check if a form element is a Lookup field.
		@param {Object} value - an value
		@return {Boolean}
		*/
		vm.isLookup = function(value) {
			if(value.type) value = value.type;
			return Array.in(value, ['search', 'multi-search', 'multi-picklist', 'lookup']);
		};

		/** @method isInfoSection
		@description check if a section contains Data Entity fields
		@param {Object} value - an value
		@return {Boolean}
		*/
		vm.isInfoSection = function(section)
		{
			if(!window.formViewSettings.isResponderOrg || !section)
				return false;

			return Array.some(section.elements, el => el.data && el.data.dataEntity);
		};

	}
};
