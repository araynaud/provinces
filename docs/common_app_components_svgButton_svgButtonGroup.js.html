/** @module common/components/svgButton

@description Buttons are used to invoke an action.<br/>
The svg-button-group component displays a Lightning button bar using svg-button components
<br/><br/>
<center>
	<img src="https://user-images.githubusercontent.com/21251436/37167250-4298d6b8-22b6-11e8-8f77-18657adbdef0.png"/>
</center>
@see {@link https://www.lightningdesignsystem.com/components/buttons/|Buttons - Lightning Design System}
@see {@link https://www.lightningdesignsystem.com/components/icons/|Icons - Lightning Design System}

@see {@link  module:common/components/svgButton|svgButton}
@see {@link  module:common/components/svgIcon|svgIcon}

@param actions          {object[]}                = list of buttons displayed.<br/>Each action button can have theses properties:<br/>text, title, click function, close, disable, show, showLoader, link | href
@param buttonClasses    {optional|String}         @   CSS classes applied to all buttons in this group

@example
<svg-button-group actions="vm.actionsLeft" class="slds-button-group" button-classes="small-button"></svg-button-group>
*/

module.exports = function (ngModule)
{
	ngModule.component('svgButtonGroup', {
		bindings: {
			actions: '=?',
			buttonClasses: '@' //CSS classes applied to all buttons in this group
		},
		template: require('./svgButtonGroup.html'),
		controllerAs: 'vm',
		controller: SvgButtonGroupController
	});

	SvgButtonGroupController.$inject = ['$window', '$element', '$attrs', '$state', 'CommonUtils', 'NavigationService'];
	function SvgButtonGroupController  ( $window,   $element,   $attrs,   $state,   CommonUtils,   NavigationService )
	{
		var vm = this;
		$window.svgButtonGroup = this;

		vm.doAction = function(action) {
			if(action && angular.isFunction(action.click)) action.click();
		};

		vm.showButton = function(action) {
			if(!action) return false;
			return CommonUtils.isMissing(action.show) || CommonUtils.getPropertyValue(action, "show") || false;
		};

		//button link url. can be constant or a function
		vm.getButtonLink = function(action) {
			return CommonUtils.getPropertyValue(action, "href") || CommonUtils.getPropertyValue(action, "link");
		};

		vm.disableButton = function(action)
		{
			return CommonUtils.getPropertyValue(action, "disabled")
				|| CommonUtils.getPropertyValue(action, "disable")
				|| CommonUtils.getPropertyValue(action, "showLoader")
				|| false;
		};
	}

};
